    <script type="text/javascript">
    var arr=[];
    function add(){
        n=Number(document.getElementById("n").value);
        arr.push(n);
        document.getElementById("list").innerHTML="The list: "+arr;   
     }
    function result(){
         var pos=0,neg=0,zero=0;
         for(var i=0;i<arr.length;i++){
             if(arr[i]>0)
                pos+=1;
            else if (arr[i]==0)
                zero+=1;
            else    
                neg+=1;
         }
            document.getElementById("result").innerHTML="POSITIVE NUMBERS : "+pos+"<br> NEGATIVE NUMBERS: "+neg+"<br>ZEROS : "+zero;

         }
